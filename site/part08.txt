    s.addEventListener('click', toggle);
    s.addEventListener('keydown', e => { if(e.key === 'Enter' || e.key === ' '){ e.preventDefault(); toggle(); } });
  });
  levelButtons.forEach(btn => btn.addEventListener('click', () => applyLevel(Number(btn.dataset.setLevel))));
  function allSet(show){

  const topHeader = document.querySelector('header.top');

  let scrollTicking = false;
  let downwardTravel = 0;
  let upwardTravel = 0;

    topHeader.classList.remove('is-hidden');
  }

  function hideTopHeader(){
    if(!topHeader.matches(':focus-within')) topHeader.classList.add('is-hidden');
  }

  function handleHeaderScroll(){
    const currentY = Math.max(0, window.scrollY);
    const delta = currentY - lastScrollY;
    const hideStart = Math.max(90, topHeader.offsetHeight * 0.8);

    if(currentY <= 16){
      showTopHeader();
      downwardTravel = 0;
      upwardTravel = 0;
    }else if(delta > 0){
      downwardTravel += delta;
      upwardTravel = 0;
      if(currentY > hideStart && downwardTravel >= 18) hideTopHeader();
    }else if(delta < 0){
      upwardTravel += -delta;
      downwardTravel = 0;
      if(upwardTravel >= 6) showTopHeader();
    }

    lastScrollY = currentY;
    scrollTicking = false;
  }

  window.addEventListener('scroll', () => {
    if(!scrollTicking){
      scrollTicking = true;
      requestAnimationFrame(handleHeaderScroll);
    }
  }, {passive:true});

  topHeader.addEventListener('focusin', showTopHeader);
  topHeader.addEventListener('pointerdown', showTopHeader);
  window.addEventListener('pageshow', () => {
    lastScrollY = Math.max(0, window.scrollY);
    if(lastScrollY <= 16) showTopHeader();
  });
